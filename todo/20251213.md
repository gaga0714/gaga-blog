from typing import Literal
import numpy as np
import ifcopenshell
from ifcopenshell.api.root import create_entity
from ifcopenshell.api.aggregate import assign_object
from ifcopenshell.api.material import assign_material
from ifcopenshell.api.geometry import assign_representation, edit_object_placement
from maas_lego.properties.pset import add_pset
from maas_lego.relationships.rel import add_filling_to_opening


def _apply_common_props(
    model: ifcopenshell.file,
    product: ifcopenshell.entity_instance,
    placement: tuple[float, float, float] = None,
    matrix: np.ndarray = None,
    space: ifcopenshell.entity_instance = None,
    storey: ifcopenshell.entity_instance = None,
    material: ifcopenshell.entity_instance = None,
    representation: ifcopenshell.entity_instance = None,
    psets: dict = None,
):
    if placement is not None:
        product.Placement = model.createIfcLocalPlacement(
            RelativePlacement=model.createIfcAxis2Placement3D(
                model.createIfcCartesianPoint(placement)
            )
        )
    if matrix is not None:
        edit_object_placement(model, product=product, matrix=matrix)
    if space is not None:
        assign_object(model, products=[product], relating_object=space)
    if storey is not None:
        assign_object(model, products=[product], relating_object=storey)
    if material is not None:
        assign_material(model, products=[product], material=material)
    if representation is not None:
        assign_representation(model, product=product, representation=representation)
    if psets is not None:
        for pset_name, pset_values in psets.items():
            add_pset(model, product=product, name=pset_name, properties=pset_values)


def create_door(
    model: ifcopenshell.file,
    name: str,
    placement: tuple[float, float, float] = None,
    matrix: np.ndarray = None,
    predefined_type: Literal[
        'DOOR', 'GATE', 'TRAPDOOR', 'USERDEFINED', 'NOTDEFINED'
    ] = 'DOOR',
    space: ifcopenshell.entity_instance = None,
    storey: ifcopenshell.entity_instance = None,
    material: ifcopenshell.entity_instance = None,
    representation: ifcopenshell.entity_instance = None,
    psets: dict = None,
    opening: ifcopenshell.entity_instance = None,
):
    door = create_entity(model, ifc_class='IfcDoor', name=name, predefined_type=predefined_type)
    _apply_common_props(
        model=model,
        product=door,
        placement=placement,
        matrix=matrix,
        space=space,
        storey=storey,
        material=material,
        representation=representation,
        psets=psets,
    )
    if opening is not None:
        add_filling_to_opening(model, element=door, opening=opening)
    return door
