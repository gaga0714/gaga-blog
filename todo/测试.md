def generate_dir_buffer_mesh(
    mesh: trimesh.Trimesh, buffer_dir: np.ndarray, depth: float
) -> Optional[trimesh.Trimesh]:
    """
    生成门窗的buffer mesh, 用于检测商品是否与门窗遮挡
    """
    if mesh is None or mesh.is_empty:
        return None
    [min_pt, max_pt] = mesh.bounds
    corners = np.array(
        [
            [min_pt[0], min_pt[1], min_pt[2]],
            [min_pt[0], min_pt[1], max_pt[2]],
            [min_pt[0], max_pt[1], min_pt[2]],
            [min_pt[0], max_pt[1], max_pt[2]],
            [max_pt[0], min_pt[1], min_pt[2]],
            [max_pt[0], min_pt[1], max_pt[2]],
            [max_pt[0], max_pt[1], min_pt[2]],
            [max_pt[0], max_pt[1], max_pt[2]],
        ]
    )
    corners_extruded = corners + depth * buffer_dir
    corners = np.vstack([corners, corners_extruded])
    new_min = np.min(corners, axis=0)
    new_max = np.max(corners, axis=0)
    new_box = trimesh.primitives.Box(bounds=[new_min, new_max])
    return new_box


def get_data_path(filename: str) -> str:
    """
    获取数据文件的正确路径，支持从项目根目录或 examples 目录运行
    """
    # 获取脚本所在目录
    script_dir = Path(__file__).parent.absolute()
    # 项目根目录（脚本的父目录）
    project_root = script_dir.parent
    
    # 尝试多个可能的路径
    possible_paths = [
        project_root / "data" / filename,  # 从项目根目录
        script_dir / ".." / "data" / filename,  # 相对路径
        Path("data") / filename,  # 当前工作目录
    ]
    
    for path in possible_paths:
        abs_path = path.resolve()
        if abs_path.exists():
            return str(abs_path)
    
    # 如果都找不到，返回第一个路径（会触发错误，但路径更清晰）
    return str(project_root / "data" / filename)
