//export
import { sqlStringToArray } from '@/common/utils';
import {
    IDataCollect,
    requestPostExportDataCollect,
    requestPostInsertOrUpdateDataCollect,
} from '@/pages/index/api/dataPlatform/collect';
import MySQLEditor from '@/pages/index/components/MySQLEditor';
import { Button, Dialog, Form, Input, toast } from '@qunhe/muya-ui';
import dayjs from 'dayjs';
import React, { ReactNode, useMemo, useState } from 'react';

const Exporter = ({
    data,
    sql,
    tableName,
    children,
}: {
    data?: IDataCollect;
    sql?: string[];
    tableName?: string;
    children: ({ show }: { show: () => void }) => ReactNode;
    onSuccess?: () => void;
}) => {
    // console.log('=== Exporter Props ===');
    // console.log('tableName:', tableName);
    // console.log('data:', data);
    // console.log('data.name:', data?.name);
    // console.log('isIfcMode:', !data && tableName);

    const [show, setShow] = useState<boolean>(false);
    const [submitLoading, setSubmitLoading] = useState<boolean>(false);
    const isIfcMode = !data && tableName;
    const finalSql = useMemo(() => {
        return isIfcMode ? `SELECT * FROM algorithm.${tableName}` : sql?.join(' ') || '';
    }, [isIfcMode, tableName, sql]);
    const finalSqlArray = useMemo(() => {
        return isIfcMode ? sqlStringToArray(finalSql) : sql || [];
    }, [isIfcMode, finalSql, sql]);

    const handleExport = async (exportPath: string) => {
        if (isIfcMode) {
            let datasetId: number;
            const newDataset = await requestPostInsertOrUpdateDataCollect({
                name: `IFC数据集_${tableName}_${dayjs().format('YYYYMMDDHHmmss')}`,
                sqlQuery: finalSql,
            });
            datasetId = newDataset.id;
            // console.log('=== datasetId ===');
            // console.log('datasetId:', datasetId);
            await requestPostExportDataCollect({
                datasetId,
                exportPath,
                sqlQuery: finalSql,
            });
        } else {
            if (!data) {
                throw new Error('数据集信息缺失');
            }
            await requestPostExportDataCollect({
                datasetId: data.id,
                exportPath,
                sqlQuery: finalSql,
            });
        }
    };

    return (
        <>
            <Dialog.Base open={show} destroyOnClose width={600}>
                <Dialog.Title onClose={() => setShow(false)}>导出数据</Dialog.Title>
                <Form<{
                    exportPath: string;
                }>
                    defaultValues={{
                        exportPath: `/seaweedfs/training/experiments/maas_data/${dayjs().format(
                            'YYYYMMDDTHHmmss',
                        )}.csv`,
                    }}
                    onSubmit={values => {
                        setSubmitLoading(true);
                        handleExport(values.exportPath)
                            .then(() => {
                                toast.success('发起导出成功');
                                setShow(false);
                            })
                            .catch(e => {
                                toast.error(`发起导出失败: ${e.message || String(e)}`);
                            })
                            .finally(() => {
                                setSubmitLoading(false);
                            });
                    }}
                    hasFeedback
                    requiredTipAlignLeft
                    labelPosition="left"
                    labelWidth={100}
                >
                    <Dialog.Content style={{ minHeight: 500 }}>
                        {isIfcMode ? (
                            <Form.Item label="表名">
                                <Input value={tableName} disabled />
                            </Form.Item>
                        ) : (
                            <Form.Item label="关联数据集">
                                <Input value={data?.name} disabled />
                            </Form.Item>
                        )}
                        <Form.Item label="导出SQL" style={{ marginTop: 12 }}>
                            <MySQLEditor defaultValue={finalSqlArray} readOnly style={{ width: 600 }} />
                        </Form.Item>
                        <Form.Item
                            label="导出路径"
                            name="exportPath"
                            rule={[{ required: true, type: 'string', message: '导出路径必填' }]}
                            style={{ marginTop: 12 }}
                        >
                            <Input width={400} />
                        </Form.Item>
                    </Dialog.Content>
                    <Dialog.Actions>
                        <Button type="normal" onClick={() => setShow(false)} loading={submitLoading}>
                            取消
                        </Button>
                        <Button type="primary" htmlType="submit" loading={submitLoading}>
                            导出
                        </Button>
                    </Dialog.Actions>
                </Form>
            </Dialog.Base>
            {children({
                show: () => setShow(true),
            })}
        </>
    );
};

export default Exporter;

//index
import { sqlArrayToString, sqlStringToArray } from '@/common/utils';
import { useQueryData } from '@/hooks/useQuery';
import {
    IDataCollect,
    requestGetDataCollectList,
    requestPostInsertOrUpdateDataCollect,
} from '@/pages/index/api/dataPlatform/collect';
import { getDataTableTree } from '@/pages/index/api/dataPlatform/dataWarehouse';
import {
    IfcSnapshotTaskStatusEnum,
    IIfcSnapshotTask,
    requestGetIfcSnapshotTaskPage,
} from '@/pages/index/api/dataPlatform/ifc';
import { LayoutContainer, TableFilterLayout } from '@/pages/index/components/MaasLayout';
import MySQLEditor from '@/pages/index/components/MySQLEditor';
import MySQLPreviewer from '@/pages/index/components/MySQLPreviewer';
import { Button, Dialog, Select, Space, toast, Typography } from '@qunhe/muya-ui';
import React, { FC, useEffect, useMemo, useRef, useState } from 'react';
import Exporter from './exporter';
import ExportRecord from './exportRecord';

const DataCollectionDetail: FC = () => {
    const [dataCollect, setDataCollect] = useState<IDataCollect>();
    const [sqlQuery, setSqlQuery] = useState<string[]>([]);
    const { id, tableName, schemaName, dataSource = 'hive', type } = useQueryData(location.hash.split('?')[1] || '');
    const that = useRef<{ sqlQuery: string[] }>({ sqlQuery: [] }).current;
    const isTable = useMemo(() => !id && tableName, [id]);
    const [ifcSnapshot, setIfcSnapshot] = useState<Array<IIfcSnapshotTask>>([]);
    const [versionIdx, setVersionIdx] = useState(0);

    useEffect(() => {
        getDataTableTree();
        if (type === 'IFC') {
            const loadingId = toast.loading('加载中');
            requestGetIfcSnapshotTaskPage({
                status: IfcSnapshotTaskStatusEnum.FINISHED,
                start: 0,
                num: 1000,
            })
                .then(res => {
                    setIfcSnapshot(res.result);
                    setVersionIdx(res.result[0]?.id);
                    const item = res.result[0]?.dataSyncJob.detail.targetMeta as any;
                    // console.log('333333:', item.tableName);
                    if (item) {
                        setSqlQuery(sqlStringToArray(`SELECT * FROM algorithm.${item.tableName}`));
                    }
                })
                .finally(() => {
                    toast.close(loadingId);
                });
        } else if (!isTable) {
            requestGetDataCollectList({ id }).then(res => {
                if (res?.[0]) {
                    setDataCollect(res[0]);
                    setSqlQuery(sqlStringToArray(res[0].sqlQuery));
                }
            });
        } else {
            setSqlQuery(sqlStringToArray(`SELECT * FROM ${schemaName}${schemaName ? '.' : ''}${tableName}`));
        }
    }, []);

    useEffect(() => {
        if (versionIdx && ifcSnapshot) {
            const item = ifcSnapshot.find(sp => sp.id === versionIdx)?.dataSyncJob.detail.targetMeta as any;
            if (item) {
                setSqlQuery(sqlStringToArray(`SELECT * FROM algorithm.${item.tableName}`));
            }
        }
    }, [versionIdx, ifcSnapshot]);

    return (
        <LayoutContainer>
            <Typography.Title level={1} style={{ marginBottom: 16 }}>
                {type === 'IFC'
                    ? 'IFC数据集'
                    : isTable
                    ? `${schemaName}${schemaName ? '.' : ''}${tableName}`
                    : dataCollect?.name}
            </Typography.Title>
            {ifcSnapshot.length > 0 && (
                <>
                    <Space style={{ marginBottom: 24 }}>
                        <Button
                            type="primary"
                            onClick={() => {
                                window.open('/#/data/design2ifc');
                            }}
                        >
                            去查看ifc转换数据
                        </Button>
                        <Typography.Text>选择snapshot版本</Typography.Text>
                        <Select value={versionIdx} onChange={e => setVersionIdx(e as number)}>
                            {ifcSnapshot.map(sp => (
                                <Select.Option key={sp.id} value={sp.id}>
                                    {(sp?.dataSyncJob?.detail?.targetMeta as any)?.tableName}
                                </Select.Option>
                            ))}
                        </Select>
                        {ifcSnapshot[versionIdx]?.remark && (
                            <Typography.Text>描述:{ifcSnapshot[versionIdx].remark}</Typography.Text>
                        )}
                    </Space>
                </>
            )}
            {(ifcSnapshot.length > 0 || dataCollect) && (
                <>
                    {dataCollect && (
                        <Typography.Title level={5} style={{ marginBottom: 16 }}>
                            {dataCollect.description}
                        </Typography.Title>
                    )}
                    <TableFilterLayout style={{ alignItems: 'flex-start', justifyContent: 'space-between' }}>
                        <Space>
                            <MySQLEditor
                                key={
                                    type === 'IFC' && versionIdx
                                        ? `sql-editor-${versionIdx}-${sqlQuery.join('')}`
                                        : `sql-editor-${sqlQuery.join('')}`
                                }
                                style={{ width: '60vw', maxHeight: 100, overflowY: 'scroll' }}
                                value={sqlQuery}
                                onChange={newVal => {
                                    that.sqlQuery = [...newVal];
                                }}
                            />
                        </Space>
                        <Space>
                            <Button
                                type="primary"
                                onClick={() => {
                                    setSqlQuery(sqlStringToArray(sqlArrayToString(that.sqlQuery)));
                                }}
                            >
                                查询
                            </Button>
                            {dataCollect && (
                                <Button
                                    type="primary"
                                    onClick={() => {
                                        if (dataCollect) {
                                            Dialog.info({
                                                title: '确认保存当前sql?',
                                                text:
                                                    sqlArrayToString(that.sqlQuery) !== sqlArrayToString(sqlQuery)
                                                        ? '当前sql语句编辑后未通过查询预览结果，无法保证正确性'
                                                        : '',
                                                cancelText: '取消',
                                                confirmText: '保存',
                                                onConfirm: () => {
                                                    requestPostInsertOrUpdateDataCollect({
                                                        id: dataCollect.id,
                                                        sqlQuery: sqlArrayToString(that.sqlQuery),
                                                    })
                                                        .then(() => {
                                                            toast.success('保存成功');
                                                            setSqlQuery(
                                                                sqlStringToArray(sqlArrayToString(that.sqlQuery)),
                                                            );
                                                        })
                                                        .catch(e => {
                                                            toast.error(`保存失败: ${String(e)}`);
                                                        });
                                                    return true;
                                                },
                                            });
                                        }
                                    }}
                                >
                                    保存
                                </Button>
                            )}
                            {(() => {
                                if (type === 'IFC' && ifcSnapshot.length > 0) {
                                    const currentSnapshot = ifcSnapshot.find(sp => sp.id === versionIdx);
                                    return currentSnapshot ? (
                                        <Exporter
                                            tableName={
                                                (currentSnapshot.dataSyncJob.detail.targetMeta as any)?.tableName
                                            }
                                        >
                                            {({ show }) => (
                                                <Button type="primary" onClick={show}>
                                                    导出数据
                                                </Button>
                                            )}
                                        </Exporter>
                                    ) : null;
                                } else if (dataCollect) {
                                    return (
                                        <Exporter sql={sqlQuery} data={dataCollect}>
                                            {({ show }) => (
                                                <Button type="primary" onClick={() => show()}>
                                                    导出数据
                                                </Button>
                                            )}
                                        </Exporter>
                                    );
                                }
                                return null;
                            })()}
                            <ExportRecord>
                                <Button type="normal">导出记录</Button>
                            </ExportRecord>
                        </Space>
                    </TableFilterLayout>
                </>
            )}
            <Typography.Paragraph style={{ margin: '6px 0' }}>
                *为防止查询时间过长，展示数据限定10条
                {!isTable && (
                    <>
                        ，<strong>总条目数{dataCollect?.totalCount}</strong>。保存不同的sql即可刷新条目总数
                    </>
                )}
            </Typography.Paragraph>
            <MySQLPreviewer
                showPage
                dataSource={dataSource}
                sqlQuery={sqlQuery}
                style={{ height: isTable ? '75vh' : '60vh', maxWidth: '95vw' }}
                limit={10}
                tableName={tableName}
            />
        </LayoutContainer>
    );
};

export default DataCollectionDetail;
