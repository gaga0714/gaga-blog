            order: {
                key: 'id',
                isAsc,
            },
            pageSize: getPageSizeFromQuery(20),
            ...filterJson,
        }).then(
            (res: {
                data: CMDResult<{
                    list: DataDetail[];
                    pageNo: number;
                    pageSize: number;
                    total: number;
                }>;
            }) => {
                console.log(res.data.d, currentStack);
                if (res.data.d && res.data.d.list.length > 0) {
                    if (isAsc) {
                        currentStack = [...res.data.d.list, ...currentStack];
                        this.prefetchNextKsg();
                    } else {
                        currentStack.push(...res.data.d.list);
                    }
                    this.list = currentStack;
                }
            }
        );
    };

    showDesign = (designId: number) => {
        const { designId: current } = this.state;
        if (designId === current) {
            return;
        }
        const params = new URLSearchParams(window.location.search);
        params.set('designId', String(designId));
        history.pushState({}, '', `?${params.toString()}`);
        this.context.setDataDetail(undefined);
        this.setState({
            designId,
            displayDesignId: undefined,
            displayLevelId: undefined,
            k2dInfo: undefined,
            ksg: undefined,
            oldLabelTypeId: undefined,
            defaultTagValue: undefined,
        });
        this.getDataDetail(designId)
            .then((item: DataDetail) => {
                this.setParseDataDetail(item);
            })
            .catch((error: any) => {
                console.error(error);
                if (typeof error === 'string') {
                    toast.error(error);
                }
                if (error.message) {
                    toast.error(error.message);
                }
            });
    };

    setParseDataDetail = (item: DataDetail, refresh = false) => {
        const { designId, displayDesignId, displayLevelId, displayImportParams } =
            getDiaplayDesign(item);
        let json = null;
        let importParams = null;
        try {
            json = JSON.parse(item.cadK2dLayerInfo!) as {
                cadFileLayerDataList?: CadK2dLayerInfo[];
            };
            importParams = JSON.parse(displayImportParams!);
        } catch (e) {
            console.error(e);
        }
        const k2dInfo = json?.cadFileLayerDataList?.find((v) => {
            return !!v.k2dOssKey;
        });
        let imageInfo: ImageInfo | undefined;
        let originImageInfo: ImageInfo | undefined;
        if (importParams?.imageFile) {
            imageInfo = {
                imageFile: importParams.imageFile,
                boundingBox2D: importParams.bdboxUpdateReq?.boundingBox2D,
                scale: importParams.scale,
                offset: importParams.offset,
            };
            if (item.editImportParams && item.importParams) {
                const ip = JSON.parse(item.importParams);
                originImageInfo = {
                    imageFile: ip.imageFile,
                    boundingBox2D: ip.bdboxUpdateReq?.boundingBox2D,
                    scale: ip.scale,
                    offset: ip.offset,
                };
            }
        }
        if (!k2dInfo && !imageInfo) {
            toast.error('cad数据获取失败，当前方案可能没有cad数据');
        }
        this.list = [item];
        let oldLabelTypeId: string | undefined;
        let defaultTagValue: Record<string, any> | undefined;
        if (item.tagJson) {
            const json = JSON.parse(item.tagJson);
            if (json.result) {
                defaultTagValue = json.result;
            }
            if (json.labelTypeId) {
                oldLabelTypeId = json.labelTypeId;
            }
        }
        this.context.setDataDetail(item);
        this.setState({
            designId,
            displayDesignId,
            displayLevelId,
            k2dInfo,
            imageInfo,
            originImageInfo,
            oldLabelTypeId,
            defaultTagValue,
        });
        if (!displayDesignId) {
            return;
        }
        this.getKsg(displayDesignId, displayLevelId, refresh)
            .then((ksg) => {
                this.setState({
                    ksg,
                });
            })
            .catch((error) => {
                console.error(error);
                toast.error('方案ksg获取失败');
            });
    };

    onClickSaveTags = () => {
        if (!this.tagPanelRef.current) {
            return;
        }
        const res = this.tagPanelRef.current.getCurrentLabelData();
        console.log('onClickSaveTags', res);
        const { designId, oldLabelTypeId } = this.state;
        const displayLabelTypeId = _labelTypeId ?? oldLabelTypeId;
        if (!res || !designId || !displayLabelTypeId) {
            return;
        }
        const json = {
            labelTypeId: displayLabelTypeId,
            result: res,
        };
        updateTagJson(designId, JSON.stringify(json))

                if (res.data.c === '0') {
                    toast.success('更新标注成功');
                    this.setState({
                        oldLabelTypeId: json.labelTypeId,
                        defaultTagValue: json.result,
                    });
                    return true;
                }
                return Promise.reject(new Error(res.data.m));
            })
            .catch((error: any) => {
                toast.error(error.message);
                return false;
            });
    };

    handleMarkFinish = () => {

    private onPrev = () => {
        const idx = this.list.findIndex((i) => {
            return i.id === this.context.dataDetail?.id;
        });
        console.log('idx', idx);
        const nextIdx = idx - 1;
        if (nextIdx < 0) {
            const id = this.list[0].id;
            this.fetchMoreData(id, undefined)?.then(() => {
                const idx = this.list.findIndex((i) => {
                    return i.id === this.context.dataDetail?.id;
                });
                const nextIdx = idx - 1;
                if (nextIdx >= 0) {
                    const item = this.list[nextIdx];
                    this.showDesign(item.designId);
                } else {
                    console.log('listRef.current', this.list);
                    toast.warning('没获取到前一个');
                }
            });
        } else {
            const item = this.list[nextIdx];
            this.showDesign(item.designId);
        }
    };

    private onNext = () => {
        const idx = this.list.findIndex((i) => {
            return i.id === this.context.dataDetail?.id;
        });
        console.log('idx', idx);
        const nextIdx = idx + 1;
        if (nextIdx >= this.list.length) {
            const id = [...this.list].reverse()[0].id;
            this.fetchMoreData(undefined, id)?.then(() => {
                const idx = this.list.findIndex((i) => {
                    return i.id === this.context.dataDetail?.id;
                });
                const nextIdx = idx + 1;
                console.log('nextIdx', nextIdx, this.list);
                if (nextIdx < this.list.length) {
                    const item = this.list[nextIdx];
                    this.showDesign(item.designId);
                } else {
                    console.log('listRef.current', this.list);
                    toast.warning('没获取到下一个');
                }
            });
        } else {
            console.log('nextIdx', nextIdx, this.list);
            const item = this.list[nextIdx];
            this.showDesign(item.designId);
        }
    };

    private onExit = () => {
        const { designId, kjlUserId } = this.state;
        const { dataDetail } = this.context;
        if (dataDetail?.status === DataDetailStatus.Approved) {
            toast.warning('已通过的方案无法编辑');
            return;
        }
        if (!kjlUserId) {
            let text = '请登录beta环境酷家乐账号';
            if ((window as any).projectStage === 'dev') {
                text = '请登录sit环境酷家乐账号';
            }
            toast.warning(text);
            this.setState({
                openLogin: true,
            });

            return;
        } else {
            if (!dataDetail) {
                return;
            }
            // 初始化状态，复制方案并跳转
            if (dataDetail.status === DataDetailStatus.Initial && designId) {
                this.iframeRef.current?.contentWindow?.postMessage(
                    {
                        type: 'system',
                        name: 'copyDesign',
                        info: {
                            designId,
                            markTable: getMarkTableFromQuery(),
                        },
                    },
                    '*'
                );
            }
            // 达人画图中，且当前账号是编辑账号，直接跳转工具
            if (
                dataDetail.status === DataDetailStatus.InProgress &&
                dataDetail.obsEditDesignId &&
                dataDetail.obsEditUserId === kjlUserId
            ) {
                this.refreshDataDetail();
                window.open(
                    `${host}/cloud/tool/h5/bim?designid=${dataDetail.obsEditDesignId}&em=0`
                );
            }
        }
    };
