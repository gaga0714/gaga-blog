#### maas平台标注内容类型新增SSL类型
实现了postMessage，从maas平台向iframe里传递ssl

```ts
//platform
export default class LabelMessage{
    static init(data:any,id:string='img-labeler'){
        new LabelMessage('init',data).send(id);
    }
    static label(data:any,id:string='img-labeler'){
        new LabelMessage('label',data).send(id);
    }
    action: 'init'|'label';
    data:any;

    constructor(action:'init'|'label',data:any){
        this.action=action;
        this.data=data;
    }
    send(id:string='img-labeler'){
        const iframeEle=document.getElementById(id) as HTMLIFrameElement;
        if(iframeEle){
            iframeEle.contentWindow?.postMessage(//?. 是可选链 (Optional Chaining) 语法，表示如果 contentWindow 存在（非空），就继续执行后面的方法。
            //contentWindow 获取对 <iframe> 内部文档的 window 对象的引用
                JSON.stringify({
                    action:this.action,
                    data:this.data,
                }),//消息内容
                '*',//目标源
            );
        }
    }
}
```

constructor(...): 这是类被实例化（new LabelMessage(...)）时会执行的特殊函数。
