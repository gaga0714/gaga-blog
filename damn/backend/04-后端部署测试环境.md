import logging
from typing import Optional
import trimesh
import numpy as np
import matplotlib.pyplot as plt
from ifcopenshell import entity_instance
from shapely.plotting import plot_polygon
from shapely import MultiPoint, Polygon, Point
from maas_ifc_tool.maas_ifc_parser import MaasIfcElement
from maas_ifc_tool.utils.rules.base_rule import BaseRule
from maas_ifc_tool.entity.score_rules import RuleType, SingleRule
from maas_ifc_tool.entity.layout_json_schema import LayoutElement2d

logger = logging.getLogger(__name__)


class OccludingWindowDoorRule(BaseRule):
    """遮挡窗口或门规则类"""

    type: str = RuleType.OCCLUDING_WINDOW_DOOR
    execution_mode = "common"
    occlude_entities: set = set(["IfcFurniture", "IfcGroup", "furniture", "group"])
    allowed_entities: set = set(["IfcWindow", "IfcDoor", "window", "door"])

    def __init__(self, rule: SingleRule, extra: Optional[dict] = None):
        super().__init__(rule, extra)
        params = rule.params or {}
        buffer = params.get("buffer", 1.0)
        self.min_iou: float = params.get("min_iou", 0.0)
        self.min_intersection_area: float = params.get("min_intersection_area", 0.0)
        self.occlude_entity = params.get("entity", None)
        if self.occlude_entity is None:
            logger.warning("OccludingWindowDoor rule params must have entity")
        elif self.occlude_entity in self.allowed_entities:
            self.buffer = buffer or 0
            self._is_valid = True
        else:
            logger.warning(
                "OccludingWindowDoor rule params entity must be IfcWindow or IfcDoor"
            )

    def get_mesh_by_ifc_element(
        self, element: entity_instance | MaasIfcElement | None, extra: dict
    ) -> tuple[str, trimesh.Trimesh]:
        """根据IFC元素获取其对应的网格模型"""
        if not self._check_element_category_id(element):
            return (None, None)
        room_center = extra.get("room_center", None)
        if (
            element
            and element.is_a() == self.occlude_entity
            and room_center is not None
        ):
            direction = element.get_direction().copy()
            dw_buffered_mesh = self._generate_dir_buffer_mesh(
                element.mesh, direction, self.buffer
            )
            if dw_buffered_mesh is not None:
                return (element.guid, dw_buffered_mesh)
        elif element and element.is_a() in self.occlude_entities:
            return (element.guid, element.mesh)
        return (None, None)

    def _generate_dir_buffer_mesh(
        self, mesh: trimesh.Trimesh, buffer_dir: np.ndarray, depth: float
    ) -> Optional[trimesh.Trimesh]:
        """
        生成向指定方向和反向扩展的buffer mesh, 用于检测商品是否与指定方向的物体遮挡
        """
        if mesh is None or mesh.is_empty:
            return None
        [min_pt, max_pt] = mesh.bounds
        corners = np.array(
            [
                [min_pt[0], min_pt[1], min_pt[2]],
                [min_pt[0], min_pt[1], max_pt[2]],
                [min_pt[0], max_pt[1], min_pt[2]],
                [min_pt[0], max_pt[1], max_pt[2]],
                [max_pt[0], min_pt[1], min_pt[2]],
                [max_pt[0], min_pt[1], max_pt[2]],
                [max_pt[0], max_pt[1], min_pt[2]],
                [max_pt[0], max_pt[1], max_pt[2]],
            ]
        )
        corners_extruded_forward = corners + depth * buffer_dir
        corners_extruded_backward = corners - depth * buffer_dir
        corners = np.vstack(
            [corners, corners_extruded_forward, corners_extruded_backward]
        )
        new_min = np.min(corners, axis=0)
        new_max = np.max(corners, axis=0)
        new_box = trimesh.primitives.Box(bounds=[new_min, new_max])
        return new_box

    def get_eneities(self):
        return set([self.occlude_entity])

    def _gen_dir_buffer_polygon(
        self, polygon: Polygon, buffer_dir: np.ndarray, depth: float
    ) -> Polygon:
        if depth <= 0:
            return polygon
        minx, miny, maxx, maxy = polygon.bounds
        corners = np.array(
            [
                [minx, miny],
                [minx, maxy],
                [maxx, miny],
                [maxx, maxy],
            ]
        )
        dir_norm = np.linalg.norm(buffer_dir)
        if dir_norm == 0:
            logger.warning("buffer_dir is zero vector")
            return None
        offset = (buffer_dir / dir_norm) * depth
        shifted_forward = corners + offset
        shifted_backward = corners - offset
        all_pts = np.vstack([corners, shifted_forward, shifted_backward])
        return MultiPoint(all_pts).convex_hull

    def _get_polygon_2d(self, element: LayoutElement2d) -> Optional[Polygon]:
        return self._get_element_polygon_2d(element)

    def _ensure_direction_inward(
        self,
        direction: np.ndarray,
        dw_polygon,
        room_polygon: Polygon,
        step: float = 0.1,
        max_step: int = 20,
    ):
        """确保方向指向房间内"""
        if np.linalg.norm(direction) == 0:
            return direction
        direction = direction / np.linalg.norm(direction)
        if room_polygon is None:
            return direction
        dw_center = dw_polygon.centroid
        for i in range(max_step):
            test_pt = Point(
                dw_center.x + direction[0] * step * i,
                dw_center.y + direction[1] * step * i,
            )
            if room_polygon.contains(test_pt):
                return direction
        return -direction

    def _get_dw_inward_dir(
        self,
        dw_polygon: Polygon,
        room_polygon: Polygon,
        step: float = 0.1,
        max_step: int = 20,
    ) -> np.ndarray:
        """获取门窗向房间内的方向"""
        if dw_polygon is None or room_polygon is None:
            return np.array([-1.0, 0.0])
        minx, miny, maxx, maxy = dw_polygon.bounds
        mid_x = (minx + maxx) / 2
        mid_y = (miny + maxy) / 2

        candidates = [
            (np.array([-1.0, 0.0]), np.array([minx, mid_y])),
            (np.array([1.0, 0.0]), np.array([maxx, mid_y])),
            (np.array([0.0, -1.0]), np.array([mid_x, miny])),
            (np.array([0.0, 1.0]), np.array([mid_x, maxy])),
        ]
        for i in range(max_step):
            for normal, pt in candidates:
                dist = step * i
                test_pt = Point(pt + normal * dist)
                if room_polygon.contains(test_pt):
                    return normal
        # fallback
        dw_center = dw_polygon.centroid
        room_center = room_polygon.centroid
        room_dw_dir = np.array(room_center) - np.array(dw_center)
        room_dw_dir = room_dw_dir / np.linalg.norm(room_dw_dir)
        best_normal = candidates[0][0]
        best_dot = -2.0
        for normal, _ in candidates:
            dot = np.dot(room_dw_dir, normal)
            if dot > best_dot:
                best_dot = dot
                best_normal = normal
        return best_normal

    def get_polygon_by_element(
        self, element: LayoutElement2d, extra: dict
    ) -> tuple[str, Polygon]:
        """根据元素获取其对应的多边形"""
        if not self._check_element_category_id(element):
            return (None, None)
        room_polygon: Polygon = extra.get("room_polygon", None)
        if element.element_type == self.occlude_entity:
            dw_polygon = self._get_polygon_2d(element)
            if dw_polygon is not None:
                inward_dir = (
                    np.array(element.element_direction)[:2]
                    if element.element_direction is not None
                    else self._get_dw_inward_dir(dw_polygon, room_polygon)
                )
                buffered_polygon = self._gen_dir_buffer_polygon(
                    dw_polygon, inward_dir, self.buffer
                )
                if buffered_polygon is not None:
                    return (element.element_id, buffered_polygon)
        elif element.element_type in self.occlude_entities:
            polygon = self._get_polygon_2d(element)
            if polygon is not None:
                return (element.element_id, polygon)
        return (None, None)

    def gen_penalty_record(self, extra: dict) -> dict:
        """格式化评分记录，添加规则类型"""
        level_id: str = extra.get("level_id", None)
        room_id: str = extra.get("room_id", None)
        polygon_detail: dict = extra.get("polygon_detail", None)
        element1: MaasIfcElement | LayoutElement2d | None = extra.get("element1", None)
        element2: MaasIfcElement | LayoutElement2d | None = extra.get("element2", None)
        room_polygon: Polygon = extra.get("room_polygon", None)
        file_path: Optional[str] = None
        if (
            self.output_img
            and isinstance(element1, LayoutElement2d)
            and isinstance(element2, LayoutElement2d)
            and room_polygon
        ):
            plt.plot(*room_polygon.exterior.xy)
            _, element1_polygon = self.get_polygon_by_element(element1, extra)
            _, element2_polygon = self.get_polygon_by_element(element2, extra)
            if element1_polygon and element2_polygon:
                e1x, e1y = element1_polygon.centroid.coords[0]
                e2x, e2y = element2_polygon.centroid.coords[0]
                if e1x >= e2x:
                    ha1 = "left"
                    ha2 = "right"
                else:
                    ha1 = "right"
                    ha2 = "left"
                if e1y >= e2y:
                    va1 = "bottom"
                    va2 = "top"
                else:
                    va1 = "top"
                    va2 = "bottom"
                plt.plot(*element1_polygon.exterior.xy)
                if element1.element_type == self.occlude_entity:
                    plt.plot(
                        *Polygon(element1.element_extrusion.polygon).exterior.xy,
                        color="red",
                    )
                height = element1.element_extrusion.height
                position_z = element1.element_extrusion.position_z
                plt.text(
                    *element1_polygon.centroid.coords[0],
                    f"{element1.element_id}-h:{height:.2f}-z:{position_z:.2f}",
                    ha=ha1,
                    va=va1,
                )
                plt.plot(*element2_polygon.exterior.xy)
                if element2.element_type == self.occlude_entity:
                    plt.plot(
                        *Polygon(element2.element_extrusion.polygon).exterior.xy,
                        color="red",
                    )
                height = element2.element_extrusion.height
                position_z = element2.element_extrusion.position_z
                plt.text(
                    *element2_polygon.centroid.coords[0],
                    f"{element2.element_id}-h:{height:.2f}-z:{position_z:.2f}",
                    ha=ha2,
                    va=va2,
                )
                intersection_polygon = element1_polygon.intersection(element2_polygon)
                if intersection_polygon:
                    plot_polygon(intersection_polygon, add_points=False)
            plt.axis("equal")
            title = f"{element1.element_id}_{element2.element_id}"
            if self.upload_to_cos:
                file_path = self._upload_to_cos(f"{self.key}_{self._md5(title)}.jpg")
            else:
                file_path = self.output_dir / f"{title}.jpg"
                self._save_to_file(file_path)
            plt.clf()
        record = {
            "name": self.name,
            "type": self.type,
            "penalty": self.weight,
            "detail": {
                "levelId": level_id,
                "roomId": room_id,
                "element1": self._get_element_info(element1),
                "element2": self._get_element_info(element2),
                "file_path": str(file_path) if file_path else None,
            },
        }
        if polygon_detail:
            record["detail"].update(polygon_detail)
        return record

    def match_result(self, extra):
        element1_type = extra.get("element1_type", None)
        element2_type = extra.get("element2_type", None)
        if (
            element1_type == self.occlude_entity
            and element2_type in self.occlude_entities
        ) or (
            element2_type == self.occlude_entity
            and element1_type in self.occlude_entities
        ):
            element1 = extra.get("element1", None)
            if isinstance(element1, LayoutElement2d):
                element2: LayoutElement2d = extra.get("element2", None)
                if (
                    not self._check_element_category_id(element1)
                    or not self._check_element_category_id(element2)
                    or not self._check_category_pair(
                        element1.category_id, element2.category_id
                    )
                ):
                    return False
                detail = extra.get("detail", None)
                if not element2 or not detail:
                    return False
                iou = detail.get("iou", 0.0)
                intersection_area = detail.get("intersection_area", 0.0)
                if iou < self.min_iou or intersection_area < self.min_intersection_area:
                    return False
                height1 = element1.element_extrusion.height
                position_z1 = element1.element_extrusion.position_z
                height2 = element2.element_extrusion.height
                position_z2 = element2.element_extrusion.position_z
                if (
                    height1 + position_z1 < position_z2
                    or height2 + position_z2 < position_z1
                ):
                    return False
            return True
        return False
