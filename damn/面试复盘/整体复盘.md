// 普通
function fn(n){
    if(n==1){
        return 1;
    }
    return n*fn(n-1);
}
// 尾递归
function fn(n, total = 1){
    if (n === 1) return total;
    return fn(n - 1, total * n);
}
// 用队列实现
function factorialQueue(n) {
    const queue = [n];
    let result = 1;
    while (queue.length > 0) {
        const x = queue.shift();

        if (x > 1) {
            result *= x;
            queue.push(x - 1);
        }
    }
    return result;
}
